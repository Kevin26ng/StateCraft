# Training controls
TRAIN_MODE = "ppo"   # one of: ppo, standard, curriculum
PPO_EPISODES = 2000
STANDARD_EPISODES = 500
CURRICULUM_EPISODES = 500
SCENARIO = "pandemic"

# Optional: mount Drive to persist outputs
USE_DRIVE = True
DRIVE_RUN_ROOT = "/content/drive/MyDrive/StateCraft_runs"

print("Config loaded. TRAIN_MODE=", TRAIN_MODE)

import os, sys, json, shutil, subprocess, pathlib

os.chdir(WORKDIR)
run([sys.executable, "-m", "pip", "install", "-q", "--upgrade", "pip"])
run([sys.executable, "-m", "pip", "install", "-q", "-r", "requirements.txt"])

try:
    import torch
    print("CUDA available:", torch.cuda.is_available())
    if torch.cuda.is_available():
        print("GPU:", torch.cuda.get_device_name(0))

# ===== 3) TRAIN =====
import time
from datetime import datetime

ts = datetime.utcnow().strftime('%Y%m%d_%H%M%S')
RUN_DIR = os.path.join(RUN_ROOT, f"run_{ts}_{TRAIN_MODE}")
os.makedirs(RUN_DIR, exist_ok=True)
print("Run dir:", RUN_DIR)

if TRAIN_MODE == "ppo":
    from training.ppo_trainer import PPOTrainer

    ckpt_dir = os.path.join(RUN_DIR, "ppo_checkpoints")
    os.makedirs(ckpt_dir, exist_ok=True)

    trainer = PPOTrainer(
        config={"scenario": SCENARIO, "num_episodes": PPO_EPISODES},
        use_wandb=False,
        checkpoint_dir=ckpt_dir,
    )
    history = trainer.train(num_episodes=PPO_EPISODES)

    with open(os.path.join(RUN_DIR, "summary.json"), "w", encoding="utf-8") as f:
        json.dump({
            "mode": "ppo",
            "episodes": len(history),
            "final": history[-1] if history else {},
            "run_dir": RUN_DIR
        }, f, indent=2)

elif TRAIN_MODE == "standard":
    cmd = [
        sys.executable, "-m", "scripts.train_cloud",
        "--episodes", str(STANDARD_EPISODES),
        "--scenario", SCENARIO,
        "--output-dir", RUN_DIR,
    ]
    run(cmd)

elif TRAIN_MODE == "curriculum":
    cmd = [
        sys.executable, "-m", "scripts.train_cloud",
        "--curriculum",
        "--episodes", str(CURRICULUM_EPISODES),
        "--scenario", SCENARIO,
        "--output-dir", RUN_DIR,
    ]
    run(cmd)

else:
    raise ValueError("TRAIN_MODE must be one of: ppo, standard, curriculum")

print("Training complete.")

# ===== 4) OPTIONAL GENERALIZATION EVAL (PPO ONLY) =====
if TRAIN_MODE == "ppo":
    from eval.generalization import run_generalization_test

    final_ckpt = os.path.join(RUN_DIR, "ppo_checkpoints", "policy_final.pt")
    if os.path.exists(final_ckpt):
        gen = run_generalization_test(final_ckpt)
        with open(os.path.join(RUN_DIR, "generalization_results.json"), "w", encoding="utf-8") as f:
            json.dump(gen, f, indent=2)
        print("Saved generalization_results.json")
    else:
        print("No policy_final.pt found, skipping eval.")
else:
    print("Skipping eval for non-PPO mode.")
